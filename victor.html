def sim_dis (p, x, M):
    y = []
    for i in range(0, M):
        u = np.random.uniform(0, 1)
        j = 1
        while ( u > np.sum(p[:j]) ):
            j += 1
        y.append(x[j-1])
    return y

def simulation_marche(N=100, S0=5, p=0.5, M=1000):
    R = np.zeros((N + 1,M))
    R[0] = S0
    for n in range(1, N + 1):
        sim = sim_dis([p, 1-p], [-1, 1], M)
        R[n] = R[n - 1] + sim
    return R


def simulation_visuelle_trajectoire(N2= 100, M2=10):
    arr = simulation_marche(M=M2)
    for i in range(0, M2):
        plt.plot(np.linspace(0, N2, N2+1), arr[:,i])
    plt.show()

Détermination de l'intervalle de confiance¶

Pour dterminer l'intervalle de confiance 95% de notre estimation $S_N \ge 5$, on remarque que:

  • En posant une fonction indicatrice $\forall k \in [1; M], Y_k = \mathbb{1}_{S_N^k \ge 5}$, alors $$ Y_k = \left\{ {1, \text{si } S_N^k \ge 5 \atop 0, \text{sinon}}\right. $$
  • Or, la probabilit qu'on recherche est $\mathbb{P}(S_N^k \ge 5) = p_k$, et par proprit de la fonction indicatrice,

$\mathbb{E}(Y_k) = p$ et $Y_k \sim \mathcal{B}(1, p_k)$.

  • De plus, les $Y_k$ tant iid, tous les $p_k$ sont gaux, donc on a $\forall k \in [1, M], p_k = p$ et
$$ \forall k \in [1, M], Y_k \sim \mathcal{B}(1, p) $$

.

Etudions $\hat Y_M = \frac{1}{M}\sum_{k=1}^{M}(Y_k)$

  • On a $\mathbb{E}(\hat Y_M) = \frac{1}{M}\sum_{k=1}^{M}\mathbb{E}(Y_k) =\frac{1}{M}\sum_{k=1}^{M} \mathbb{P}(S_N^k \ge 5) = \mathbb{P}(S_N^k \ge 5) = p$ car les $S_N^k$ sont iid. D'o $\hat Y_M$ est sans biais.
  • On a $\mathbb{V}(\hat Y_M) = \frac{1}{M^2} * M *\mathbb{V}(Y_1) = \frac{1}{M} * [ \mathbb{E}(\mathbb{1}_{S_N^k \ge 5})^2 - (\mathbb{E}(\mathbb{1}_{S_N^k \ge 5}))^2] = \frac{1}{M}*p*(1-p)$

D'aprs le thorme central limite: $$ \sqrt{\frac{M}{p(1-p)}}|\hat Y_M - p | \sim \mathcal{N}(0, 1) $$

Donc par proprit sur les quantiles de la loi normale centre rduite, en posant $q_{1 -\alpha} = - q_{\alpha} = 1.96$, avec $\alpha = 0.05$, et on a:

  • $\mathbb{P} (-1.96 \le \sqrt{\frac{M}{p(1-p)}}\hat Y_M - p \le 1.96) = 1 - \alpha = 95\%$
  • Or $\forall p \in [0, 1], p(1-p) \in [0, 1/4]$, d'o $\forall p \in [0, 1], 0 \le \sqrt{\frac{M}{p(1-p)}} \le \frac{1}{2\sqrt{M}}$ d'o $\mathbb{P} (-1.96 * \frac{1}{2\sqrt{M}} \le \hat Y_M - p \le 1.96 * \frac{1}{2\sqrt{M}}) = 1 - \alpha = 95\%$ ie $$ \mathbb{P} (\hat Y_M -1.96 * \frac{1}{2\sqrt{M}} \le p \le \hat Y_M + 1.96 * \frac{1}{2\sqrt{M}}) = 95\% $$

Un intervalle de confiance 95% de notre estimation est donc $[\hat Y_M \pm 1.96 * \frac{1}{2\sqrt{M}}]$

def Q4_chaine_markov(N=3, M=1000):
    P = np.random.random((10, 10))
    P = P/np.sum(P, axis=1)[:, None] # Normalisation des lignes, pour avoir une matrice stochastique
    Mat = np.zeros((N + 1, M), dtype=int)
    Mat[0] = 1
    for n in range(1, N + 1):
        for m in range(0, M):
            s = sim_dis(P[(Mat[n-1, m]) - 1], np.linspace(1, 10, 10, dtype=int), 1)
            Mat[n, m] = s[0]
    return Mat

def death_time(Ne=5, Ns=10, P=P7, Q=P7M, N=100):
    Nb_tour = 1
    Nb_vivant = Ne
    Mat = np.zeros((N + 1, Ne), dtype=int)
    Mons = [Ns]
    Mat[0] = 1
    while (Nb_vivant > 0 and Nb_tour < N):
        s = sim_dis(Q[(Mons[-1]) - 1], np.linspace(1, 10, 10, dtype=int), 1) # Le monstre se dplace
        Mons.append(s[0])
        for m in range(0, Ne):
            if (Mat[Nb_tour-1, m] != -1):
                s = sim_dis(P[(Mat[Nb_tour-1, m]) - 1], np.linspace(1, 10, 10, dtype=int), 1)
                Mat[Nb_tour, m] = s[0]
                if (Mat[Nb_tour, m] == Mons[-1]):
                    Mat[Nb_tour, m] = -1
            else:
                Mat[Nb_tour, m] = -1
        Nb_vivant = sum(Mat[Nb_tour] != -1)
        Nb_tour += 1
    return Nb_tour - 1 # Car le premier tour n'est pas compt, c'est l'initialisation
